"tenant"."qqqqqqqqqqqqqqqqqqqqqg$6_group1" data 
    INNER JOIN "tenant"."qqqqqqqqqqqqqqqqqqqqqg$6_group1_enablement" enablement ON data.interview_id = enablement.interview_id
    INNER JOIN "tenant"."qqqqqqqqqqqqqqqqqqqqqg$6_group1_validity" validity ON data.interview_id = validity.interview_id
 WHERE data.interview_id = ANY(@ids)
