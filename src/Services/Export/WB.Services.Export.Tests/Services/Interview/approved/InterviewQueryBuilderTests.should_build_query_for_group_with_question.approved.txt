select data.interview__id as data__interview_id ,  data."num1" as "data__num1" ,  enablement."group1" as "enablement__group1"
,  enablement."num1" as "enablement__num1" ,  validity."num1" as "validity__num1"  from 
"qqqqqqqqqqqqqqqqqqqqqg$6_quest" data 
 INNER JOIN     "qqqqqqqqqqqqqqqqqqqqqg$6_quest_e" enablement ON data.interview__id = enablement.interview__id
 INNER JOIN     "qqqqqqqqqqqqqqqqqqqqqg$6_quest_v" validity ON data.interview__id = validity.interview__id
WHERE validity.interview__id = ANY(@ids)
